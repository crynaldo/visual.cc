-- Initialize Visual
local visual = require("visual")

-- Configure settings
visual.config({
    bypass = "quantum",
    autoExecute = true,
    debugMode = false
})

-- Load script from file or URL
local script = visual.loadScript("https://example.com/script.lua")

-- Execute script with advanced options
local success, result = visual:execute(script, {
    silent = true,
    timeout = 5000,
    environment = "protected"
})

-- Handle execution result
if success then
    visual.notify("Script executed successfully!")
    return result
else
    visual.error("Execution failed: " .. result)
end
